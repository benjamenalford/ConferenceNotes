Variable scoping

let and const

  • scoped to curly braces, not to the function

Variable scoping

ES5: var

function hello() {
    console.log(message); // undefined
    if (true) {
        var message = 'hello world';
        console.log(message); // 'hello world'
    }
    console.log(message); // 'hello world'
}

ES6: let, const

function hello() {
    console.log(message); // exception
    if (true) {
        let message = 'hello world';
        console.log(message); // 'hello world'
    }
    console.log(message); // exception
}

"fat arrow" functions

=>

  • binds "this"
  • always anonymous
  • implicit last line return
  • function is still best for heavy lifting

"fat arrow" functions

ES5: function

var array = [1,2,3];
array.map(function (a) { return a * 2; });

ES6: =>

var array = [1,2,3];
array.map(a => a * 2);

String Templates

  • Use variables inline
  • Multi-line strings
  • Automatically encodes

String Templates

ES5: add strings

var a = 5;
var b = 10;
var message = "Fifteen is " + (a + b) + " and\n";
message += "not " + (2 * a + b) + ".");
console.log(message);
// "Fifteen is 15 and
// not 20."

ES6: string templates

var a = 5;
var b = 10;
var message = (`Fifteen is ${a + b} and
not ${2 * a + b}.`;
console.log(message);
// "Fifteen is 15 and
// not 20."
